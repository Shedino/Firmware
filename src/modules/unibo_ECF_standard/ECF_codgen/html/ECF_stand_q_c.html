  437      *  Inport: '<Root>/K_i'
  438      *  MATLAB Function: '<Root>/state update'
  439      *  MATLAB Function: '<S6>/q_dot'
  440      */
  441     t = (ECF_stand_q_U.Gyro[0] - rtb_DiscreteTimeIntegrator1_idx) +
  442       ECF_stand_q_U.K_i * rtb_Divide_idx_0;
  443     absxk = (ECF_stand_q_U.Gyro[1] - rtb_DiscreteTimeIntegrator1_i_0) +
  444       ECF_stand_q_U.K_i * rtb_Divide_idx;
  445     scale = (ECF_stand_q_U.Gyro[2] - rtb_DiscreteTimeIntegrator1_i_1) +
  446       ECF_stand_q_U.K_i * scale;
  447   
  448     /* Gain: '<S6>/cons' incorporates:
  449      *  Constant: '<S6>/Constant'
  450      *  MATLAB Function: '<S6>/q_dot'
  451      *  SignalConversion: '<S10>/TmpSignal ConversionAt SFunction Inport2'
  452      */
  453     /* MATLAB Function 'state Integration /q_dot': '<S10>:1' */
  454     /* quatmultiply(q1,q2) calculates the quaternion product, n, for two given */
  455     /* quaternions, q1 and q2; */
  456     /* '<S10>:1:8' */
  457     /* '<S10>:1:9' */
  458     /* '<S10>:1:10' */
  459     /* '<S10>:1:11' */
  460     /* '<S10>:1:13' */
  461     ECF_stand_q_B.cons[0] = (rtb_y[0] * ECF_stand_q_P.Constant_Value_g - ((rtb_y[1]
  462       * t + rtb_y[2] * absxk) + rtb_y[3] * scale)) * 0.5 * ECF_stand_q_P.cons_Gain;
  463     ECF_stand_q_B.cons[1] = ((rtb_y[0] * t + ECF_stand_q_P.Constant_Value_g *
  464       rtb_y[1]) + (rtb_y[2] * scale - rtb_y[3] * absxk)) * 0.5 *
  465       ECF_stand_q_P.cons_Gain;
  466     ECF_stand_q_B.cons[2] = ((rtb_y[0] * absxk + ECF_stand_q_P.Constant_Value_g *
  467       rtb_y[2]) + (rtb_y[3] * t - rtb_y[1] * scale)) * 0.5 *
  468       ECF_stand_q_P.cons_Gain;
  469     ECF_stand_q_B.cons[3] = ((rtb_y[0] * scale + ECF_stand_q_P.Constant_Value_g *
  470       rtb_y[3]) + (rtb_y[1] * absxk - rtb_y[2] * t)) * 0.5 *
  471       ECF_stand_q_P.cons_Gain;
  472   }

  474   /* Model update function for TID1 */
  475   static void ECF_stand_q_update1(void)  /* Sample time: [0.02s, 0.0s] */
  476   {
  477     /* Update for DiscreteIntegrator: '<S6>/Discrete-Time Integrator' */
  478     ECF_stand_q_DW.DiscreteTimeIntegrator_DSTATE[0] +=
  479       ECF_stand_q_P.DiscreteTimeIntegrator_gainval * ECF_stand_q_B.cons[0];
  480     ECF_stand_q_DW.DiscreteTimeIntegrator_DSTATE[1] +=
  481       ECF_stand_q_P.DiscreteTimeIntegrator_gainval * ECF_stand_q_B.cons[1];
  482     ECF_stand_q_DW.DiscreteTimeIntegrator_DSTATE[2] +=
  483       ECF_stand_q_P.DiscreteTimeIntegrator_gainval * ECF_stand_q_B.cons[2];
  484     ECF_stand_q_DW.DiscreteTimeIntegrator_DSTATE[3] +=
  485       ECF_stand_q_P.DiscreteTimeIntegrator_gainval * ECF_stand_q_B.cons[3];
  486   
  487     /* Update for DiscreteIntegrator: '<S6>/Discrete-Time Integrator1' */
  488     ECF_stand_q_DW.DiscreteTimeIntegrator1_DSTATE[0] +=
  489       ECF_stand_q_P.DiscreteTimeIntegrator1_gainval * ECF_stand_q_B.bias_dot[0];
  490     ECF_stand_q_DW.DiscreteTimeIntegrator1_DSTATE[1] +=
  491       ECF_stand_q_P.DiscreteTimeIntegrator1_gainval * ECF_stand_q_B.bias_dot[1];
  492     ECF_stand_q_DW.DiscreteTimeIntegrator1_DSTATE[2] +=
  493       ECF_stand_q_P.DiscreteTimeIntegrator1_gainval * ECF_stand_q_B.bias_dot[2];
  494   
  495     /* Update absolute time */
  496     /* The "clockTick1" counts the number of times the code of this task has
  497      * been executed. The absolute time is the multiplication of "clockTick1"
  498      * and "Timing.stepSize1". Size of "clockTick1" ensures timer will not
  499      * overflow during the application lifespan selected.
  500      * Timer of this task consists of two 32 bit unsigned integers.
  501      * The two integers represent the low bits Timing.clockTick1 and the high bits
  502      * Timing.clockTickH1. When the low bit overflows to 0, the high bits increment.
  503      */
  504     if (!(++ECF_stand_q_M->Timing.clockTick1)) {
  505       ++ECF_stand_q_M->Timing.clockTickH1;
  506     }
  507   
  508     ECF_stand_q_M->Timing.t[1] = ECF_stand_q_M->Timing.clockTick1 *
  509       ECF_stand_q_M->Timing.stepSize1 + ECF_stand_q_M->Timing.clockTickH1 *
  510       ECF_stand_q_M->Timing.stepSize1 * 4294967296.0;
  511   }
